def rolling_window(seq, size, step):
    return (seq[pos:pos+size] for pos in range(0, len(seq)-size+step, step))

data = []
for s in rolling_window(aen, 1000, 100):
   data.append(metrics.chisq(s, aen, cf.pattern_counter_raw).statistic)
